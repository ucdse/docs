package "Flask App Subsystem" {
    
  package "models (Domain Entities)" {
    class Station {
      __tablename__ = "station"
      +number : Integer
      +name : String
    }

    class Availability {
      __tablename__ = "availability"
      +id : Integer
      +available_bikes : Integer
      +available_bike_stands : Integer
    }

    class WeatherForecast {
      __tablename__ = "weather"
      +id : Integer
      +temp : Float
      +to_dict()
    }

    class User {
      __tablename__ = "user"
      +id : Integer
      +email : String
      +password_hash : String
    }

    class Session {
      __tablename__ = "session"
      +id : Integer
      +user_id : Integer
      +token : String
      +utcnow()
    }

    class ChatHistory {
      __tablename__ = "chat_history"
      +id : Integer
      +user_id : Integer
      +message : String
      +response : String
    }
  }

  package "services (Software Internal Classes)" {
    class PredictionService {
      -rf_model : RandomForestRegressor
      +_load_model()
      +get_station_predictions(station_id: int)
    }

    class WeatherService {
      +get_weather()
    }

    class StationService {
      +list_stations()
      +get_recent_station_availability(number, lookback)
      +get_all_stations_latest_availability()
      +_availability_to_dict()
      +_station_to_dict()
    }
    
    class UserService {
      +create_access_token()
      +create_refresh_token()
      +verify_access_token()
      +verify_refresh_token()
      +login_user()
      +refresh_tokens()
      +logout_user()
      +register_user()
      +send_verification_code()
      +activate_user()
      +activate_by_token()
      +serialize_user()
    }

    class JourneyService {
      +get_matrix_durations(origins, destinations, mode="walking")
      +find_best_route(start_lat, start_lon, end_lat, end_lon)
    }

    class ChatService {
      +get_chat_history(session_id: str)
      +generate_session_id(user_id: int, chat_id: str)
      +get_session_messages(session_id: str, user_id: int)
      +generate_chat_response(session_id: str, user_message, user_id)
      +generate_chat_stream(session_id, user_message, user_id)
      +_ensure_session()
      +_generate_title()
      +_map_role()
    }
  }

  package "api (Controllers)" {
    class StationRoutes {
      +list_stations()

    class UserRoutes {

      +refresh()
      +logout()

    }
    class ChatRoutes {

      +get_session_history(session_id: str)
      +_require_auth()
    }
    class WeatherRoutes {
      +get_weather_forecast()
      +_validation_error_message()
    }
  }
}
